SELECT DISTINCT
  ?humanGene ?humanGeneLabel
  ?ratGene ?ratGeneLabel
  ?organ ?organLabel
  ?disease ?diseaseLabel
WHERE {
  # pick the organ we care about
  VALUES ?organ { wd:Q1073 }  # brain

  # human gene must be from Homo sapiens
  ?humanGene wdt:P703 wd:Q15978631 .        # P703 = found in taxon, Homo sapiens (Q15978631)

  # link gene to that organ
  ?humanGene wdt:P2293 ?disease .              # P927 = anatomical location


  # find the rat ortholog
  ?humanGene wdt:P684 ?ratGene .            # P684 = ortholog
  ?ratGene   wdt:P703 wd:Q184224 .          # rat taxon = Rattus norvegicus (Q184224)
  


  SERVICE wikibase:label {
    bd:serviceParam wikibase:language "en".
  }
}
